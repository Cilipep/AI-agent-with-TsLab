                    if (!trailActivated)
                    {
                        double unrealizedPct = (highPrice - entryPrice) / entryPrice;
                        if (unrealizedPct >= TRAIL_ACTIVATE_PCT)
                        {
                            trailActivated = true;
                            trailHigh = highPrice;
                        }
                    }
                    else
                    {
                        trailHigh = Math.Max(trailHigh, highPrice);
                    }

                    // Stop Loss
                    if (lowPrice <= slLevel)
                    {
                        pos.CloseAtMarket(i + 1, "SkSred_Close");
                        trailActivated = false;
                        continue;
                    }

                    // Take Profit
                    if (highPrice >= tpLevel)
                    {
                        pos.CloseAtMarket(i + 1, "SkSred_Close");
                        trailActivated = false;
                        continue;
                    }

                    // Trailing Stop
                    if (trailActivated && lowPrice <= trailHigh * (1 - TRAIL_DISTANCE_PCT))
                    {
                        pos.CloseAtMarket(i + 1, "SkSred_Close");
                        trailActivated = false;
                        continue;
                    }

                    // Signal exit
                    if (longExit[i])
                    {
                        pos.CloseAtMarket(i + 1, "SkSred_Close");
                        trailActivated = false;
                    }
                }

                // Entry
                bool hasPosition = posList.GetActiveForBar(i).Any();
                if (!hasPosition && longEntry[i])
                {
                    posList.BuyAtMarket(i + 1, scaledSize, "SkSred_Long");
                    trailActivated = false;
                    trailHigh = currentPrice;
                }

                result[i] = currentPrice;
            }

            return result;
        }
    }
}
